\chapter{Appendix A}

The features of peoms we wish to be able to detect:
\begin{itemize}
\item{Structure}
	\begin{itemize}
	\setlength{\itemsep}{0pt}
	\begin{multicols}{2}
	\item{Number of stanzas}
	\item{Number of lines per stanza}
	\item{Number of repeated lines}
	\columnbreak
	\item{Repeated phrases}
	\item{Repeated words}
	\item{Unusual words}
	\end{multicols}
	\end{itemize} 
\item{Rhyme}
	\begin{itemize}
	\setlength{\itemsep}{0pt}
	\begin{multicols}{2}
	\item{Rhyme scheme between lines}
	\item{Rhyme scheme within a line}
	\columnbreak
	\item{Presence of regular rhyme scheme}
	\end{multicols}
	\end{itemize}
\item{Rhythm}
	\begin{itemize}
	\setlength{\itemsep}{0pt}
	\begin{multicols}{2}
	\item{Number of syllables}
	\columnbreak
	\item{Stress pattern for each line}
	\end{multicols}
	\end{itemize}
\item{Line patterns}
	\begin{itemize}
	\setlength{\itemsep}{0pt}
	\begin{multicols}{2}
	\item{Alliteration}
	\item{Assonance}
	\columnbreak
	\item{Consonance}
	\end{multicols}
	\end{itemize}
\item{Rhetoric}
	\begin{itemize}
	\setlength{\itemsep}{0pt}
	\begin{multicols}{2}
	\item{Similes}
	\item{Metaphors}
	\columnbreak
	\item{Onomatopoeia}
	\end{multicols}
	\end{itemize}
\item{Contextual}
	\begin{itemize}
	\setlength{\itemsep}{0pt}
	\begin{multicols}{2}
	\item{Tense}
	\item{Point of view}
	\item{Characters}
	\item{Objects}
	\columnbreak
	\item{Locations}
	\item{Descriptions}
	\item{Actions}
	\item{Relationships}
	\end{multicols}
	\end{itemize}
\end{itemize}
